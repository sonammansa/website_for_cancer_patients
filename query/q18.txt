declare
v_s hospital.state%type:=&state;
begin
for v1 in(select h.hname,s.sname from hospital h,works_at w,specialists s where h.state=v_s and h.hid=w.hid and w.sid=s.sid) loop
dbms_output.put_line(chr(10)||v1.hname||chr(10)||'specialists--'||chr(10)||v1.sname);
end loop;
exception
when no_data_found then
dbms_output.put_line('no such value exists');
end;
